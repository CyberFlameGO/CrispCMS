529  public static function integrityCheck($PluginName, $PluginMetadata) {
530  $PluginFolder = \crisp\api\Config::get("plugin_dir");
531  $parsedConfigs = array();
532  $failedConfigs = array();
533  $failedChecks = array();
534  $parsedChecks = array();
535  $parsedFiles = array();
536  $failedFiles = array();
537  $integrity = true;
538  if (isset($PluginMetadata->onInstall->createKVStorageItems) && \is_object($PluginMetadata->onInstall->createKVStorageItems)) {
539  foreach ($PluginMetadata->onInstall->createKVStorageItems as $Key => $Value) {
540  if (is_array($Value) || \is_object($Value)) {
541  $Value = \serialize($Value);
542  }
543  try {
544  $KeyExists = \crisp\api\Config::exists("plugin_" . $PluginName . "_$Key");
545  $parsedConfigs[$Key] = $Value;
546  if (!$KeyExists) {
547  $failedConfigs[$Key] = $Value;
548  }
549  } catch (\PDOException $ex) {
550  $failedConfigs[$Key] = $Value;
551  }
552  }
553  $parsedChecks[] = "KVStorageItems";
554  if (count($failedConfigs) > 0) {
555  $integrity = false;
556  $failedChecks[] = "KVStorageItems";
557  }
558  }
559 
560 
561 
562  if (!self::isInstalled($PluginName)) {
563  $integrity = false;
564  $failedChecks[] = "isInstalled";
565  } else {
566  $parsedChecks[] = "isInstalled";
567  }
568 
569  $PluginPath = \realpath(__DIR__ . "/../../../../$PluginFolder" . DIRECTORY_SEPARATOR . "$PluginName") . DIRECTORY_SEPARATOR;
570  if (\file_exists($PluginPath . "sha256sum.txt")) {
571 
572  foreach (file($PluginPath . "sha256sum.txt") as $line) {
573  if (!ctype_space($line)) {
574  $line = explode(" ", $line);
575  $Hash = $line[0];
576  $File = trim($line[1]);
577  $HashFile = \hash_file("sha256", $PluginPath . $File);
578  $parsedFiles[$File] = $HashFile;
579  if ($HashFile != $Hash) {
580  $failedFiles[$File] = $HashFile;
581  }
582  }
583  }
584  if (count($failedFiles) > 0) {
585  $integrity = false;
586  $failedChecks[] = "hash_file";
587  }
588  $parsedChecks[] = "hash_file";
589  }
590 
591  return array(
592  "integrity" => $integrity,
593  "parsedConfigs" => $parsedConfigs,
594  "failedConfigs" => $failedConfigs,
595  "parsedFiles" => $parsedFiles,
596  "failedFiles" => $failedFiles,
597  "isInstalled" => self::isInstalled($PluginName),
598  );
599  }
