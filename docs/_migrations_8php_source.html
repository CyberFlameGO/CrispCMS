121  public function migrate() {
122  echo "Starting Migration..." . PHP_EOL;
123  $files = glob(__DIR__ . '/../migrations/*.{php}', GLOB_BRACE);
124  foreach ($files as $file) {
125  if (basename($file) == "template.php") {
126  continue;
127  }
128 
129  $MigrationName = substr(basename($file), 0, -4);
130 
131  if ($this->isMigrated($MigrationName)) {
132  echo "$MigrationName is already migrated, skipping!" . PHP_EOL;
133  continue;
134  }
135 
136  $Class = "\crisp\migrations\\" . explode("_", $MigrationName)[1];
137 
138  include $file;
139 
140  $Migration = new $Class();
141 
142  if ($Migration->run()) {
143 
144  $statement = $this->Database->prepare("INSERT INTO schema_migration (file) VALUES (:file)");
145 
146  $statement->execute(array(":file" => $MigrationName));
147  echo "Migrated $MigrationName" . PHP_EOL;
148  } else {
149  echo "Failed to migrate $MigrationName" . PHP_EOL;
150  }
151  }
