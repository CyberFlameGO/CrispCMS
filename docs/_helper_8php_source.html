273  public static function isSerialized($data, $strict = true) {
274  // if it isn't a string, it isn't serialized.
275  if (!is_string($data)) {
276  return false;
277  }
278  $data = trim($data);
279  if ('N;' == $data) {
280  return true;
281  }
282  if (strlen($data) < 4) {
283  return false;
284  }
285  if (':' !== $data[1]) {
286  return false;
287  }
288  if ($strict) {
289  $lastc = substr($data, -1);
290  if (';' !== $lastc && '}' !== $lastc) {
291  return false;
292  }
293  } else {
294  $semicolon = strpos($data, ';');
295  $brace = strpos($data, '}');
296  // Either ; or } must exist.
297  if (false === $semicolon && false === $brace)
298  return false;
299  // But neither must be in the first X characters.
300  if (false !== $semicolon && $semicolon < 3)
301  return false;
302  if (false !== $brace && $brace < 4)
303  return false;
304  }
305  $token = $data[0];
306  switch ($token) {
307  case 's' :
308  if ($strict) {
309  if ('"' !== substr($data, -2, 1)) {
310  return false;
311  }
312  } elseif (false === strpos($data, '"')) {
313  return false;
314  }
315  // or else fall through
316  case 'a' :
317  case 'O' :
318  return (bool) preg_match("/^{$token}:[0-9]+:/s", $data);
319  case 'b' :
320  case 'i' :
321  case 'd' :
322  $end = $strict ? '$' : '';
323  return (bool) preg_match("/^{$token}:[0-9.E-]+;$end/", $data);
324  }
325  return false;
326  }
